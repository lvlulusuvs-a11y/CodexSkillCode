```python
import asyncio
import random
import time
from dataclasses import dataclass
from typing import Generic, TypeVar, Optional

T = TypeVar("T")

@dataclass
class Result(Generic[T]):
    """Result type with error handling."""
    value: Optional[T] = None
    error: Optional[Exception] = None
    latency_ms: float = 0.0

    @property
    def is_ok(self) -> bool:
        return self.error is None

    def unwrap(self) -> T:
        if self.error:
            raise self.error
        assert self.value is not None
        return self.value

class CircuitBreaker:
    def __init__(self, threshold: int = 5, timeout: float = 30.0):
        self.threshold = threshold
        self.timeout = timeout
        self._failures = 0
        self._state = "closed"
        self._last_failure = 0.0

    async def call(self, func, *args, **kwargs):
        if self._state == "open":
            if time.monotonic() - self._last_failure > self.timeout:
                self._state = "half-open"
            else:
                raise Exception("Circuit breaker open")
        try:
            result = await func(*args, **kwargs)
            if self._state == "half-open":
                self._state = "closed"
                self._failures = 0
            return result
        except Exception as exc:
            self._failures += 1
            self._last_failure = time.monotonic()
            if self._failures >= self.threshold:
                self._state = "open"
            raise exc

async def with_retry(func, max_retries=3, backoff=1.0, max_backoff=60.0):
    for attempt in range(max_retries + 1):
        try:
            return await func()
        except Exception as exc:
            if attempt == max_retries:
                raise
            delay = min(backoff * (2 ** attempt), max_backoff)
            delay += random.uniform(0, delay * 0.1)
            print(f"Retry {attempt + 1}/{max_retries} after {delay:.2f}s: {exc}")
            await asyncio.sleep(delay)
```

```rust
use std::future::Future;
use std::time::Duration;
use tokio::time::sleep;
use rand::Rng;

#[derive(Debug)]
pub enum CircuitState {
    Closed,
    Open,
    HalfOpen,
}

pub struct CircuitBreaker {
    state: CircuitState,
    failures: u32,
    threshold: u32,
    timeout: Duration,
    last_failure: Option<std::time::Instant>,
}

impl CircuitBreaker {
    pub fn new(threshold: u32, timeout: Duration) -> Self {
        Self {
            state: CircuitState::Closed,
            failures: 0,
            threshold,
            timeout,
            last_failure: None,
        }
    }

    pub async fn call<F, Fut, T>(&mut self, f: F) -> Result<T, String>
    where
        F: FnOnce() -> Fut,
        Fut: Future<Output = Result<T, String>>,
    {
        if matches!(self.state, CircuitState::Open) {
            if let Some(last) = self.last_failure {
                if last.elapsed() > self.timeout {
                    self.state = CircuitState::HalfOpen;
                } else {
                    return Err("circuit breaker open".into());
                }
            }
        }
        match f().await {
            Ok(value) => {
                if matches!(self.state, CircuitState::HalfOpen) {
                    self.state = CircuitState::Closed;
                    self.failures = 0;
                }
                Ok(value)
            }
            Err(e) => {
                self.failures += 1;
                self.last_failure = Some(std::time::Instant::now());
                if self.failures >= self.threshold {
                    self.state = CircuitState::Open;
                }
                Err(e)
            }
        }
    }
}

pub async fn with_retry<F, Fut, T>(
    mut f: F,
    max_retries: u32,
    base_delay: Duration,
    max_delay: Duration,
) -> Result<T, String>
where
    F: FnMut() -> Fut,
    Fut: Future<Output = Result<T, String>>,
{
    let mut rng = rand::thread_rng();
    for attempt in 0..=max_retries {
        match f().await {
            Ok(value) => return Ok(value),
            Err(e) if attempt < max_retries => {
                let delay = base_delay * 2u32.pow(attempt);
                let delay = delay.min(max_delay);
                let jitter = Duration::from_millis(
                    rng.gen_range(0..delay.as_millis() as u64 / 10)
                );
                sleep(delay + jitter).await;
            }
            Err(e) => return Err(e),
        }
    }
    unreachable!()
}
```
